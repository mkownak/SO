#!/bin/bash

# Funkcja do obliczania narastającej sumy czasów trwania procesów
oblicz_sume_czasow() {
    plik="$1"
    suma=0
    liczba_procesow=0

    while read -r linia; do
        czas_trwania=$(echo "$linia" | awk '{print $2}')
        suma=$((suma + czas_trwania))
        liczba_procesow=$((liczba_procesow + 1))
    done < "$plik"

    echo "Suma czasów: $suma / Liczba procesów: $liczba_procesow"
}

# Wywołanie funkcji dla różnych plików
oblicz_sume_czasow "procesy"
oblicz_sume_czasow "procesy_dlugotrwale"
oblicz_sume_czasow "procesy_krotkotrwale"
